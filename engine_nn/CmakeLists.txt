cmake_minimum_required(VERSION 3.16)
project(cleanr-engine VERSION 0.1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_compile_options(-Wall -Wextra -O2)

add_executable(cleanr-engine
    src/main.cpp
    src/nn/matrix.cpp
    src/nn/layer.cpp
    src/nn/network.cpp
    src/nn/autoencoder.cpp
)

target_include_directories(cleanr-engine PRIVATE src)
